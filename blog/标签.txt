<script type="text/javascript">

    function OnInput (event) {
        document.getElementById('tag-button').innerHTML = document.getElementById('tag-content').value;
    }

    function changeText() {
        if (document.getElementById('tag-content').value === "") {
            alert('标签内容不能为空！');
            document.getElementById('tag-content').value = "default";
            document.getElementById('tag-button').innerHTML = "default";
        }
    }

    function selectStyle() {
        document.getElementById('tag-button').className = "btn " + document.getElementById('tag-style').value;
    }

    function changeTag(id) {
        alert(document.getElementById(id));
        var button = document.getElementById(id);
        document.getElementById('module-content').value = button.value;
        document.getElementById('module-button').className = document.getElementById('tag-button').className;
        alert(document.getElementById('myModal'));
    }
